# 思路

**费马小定理、逆元**

A%9973 = n 求 (A/B)%9973 = (A*B`)%997s = (n*B`%9973)%9973

因为9973是素数，正好可以用费马小定理，用快速幂求逆元

# 代码

```cpp
#include <iostream>
#define mod(x, m) ((x) % (m) + (m)) % (m)
#define endl '\n'
using namespace std;
typedef long long ll;
ll powm(ll x, ll n) {
    ll res = 1;
    while (n) {
        if (n&1) res = res * x % 9973;
        x = x * x % 9973;
        n >>= 1;
    }
    return res;
}
int main() {
    int t;
    cin >> t;
    ll n, b;
    while (t--) {
        cin >> n >> b;
        cout << (n*powm(b, 9971)%9973) << endl;
    }
    return 0;
}
```
